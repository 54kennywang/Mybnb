/**
     * Computes a salted PBKDF2 hash of given plaintext password, empty passwords are not supported.
     * @param password user actual password
     * @return hashed password
     */
    public static String getSaltedHash(String password)


/**
     * Checks whether user actual password corresponds to a stored salted hash password.
     * @param password user actual password
     * @param stored salted hash password
     * @return true if they correspond; false otherwise
     */
    public static boolean check(String password, String stored)
